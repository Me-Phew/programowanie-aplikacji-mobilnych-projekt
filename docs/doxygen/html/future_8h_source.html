61class FutureHandle {
62 public:
64 FutureHandle();
65 explicit FutureHandle(FutureHandleId id) : FutureHandle(id, nullptr) {}
66 FutureHandle(FutureHandleId id, detail::FutureApiInterface* api);
67 ~FutureHandle();
68
69 // Copy constructor and assignment operator.
70 FutureHandle(const FutureHandle& rhs);
71 FutureHandle& operator=(const FutureHandle& rhs);
72
73#if defined(FIREBASE_USE_MOVE_OPERATORS)
74 // Move constructor and assignment operator.
75 FutureHandle(FutureHandle&& rhs) noexcept;
76 FutureHandle& operator=(FutureHandle&& rhs) noexcept;
77#endif // defined(FIREBASE_USE_MOVE_OPERATORS)
78
79 // Comparison operators.
80 bool operator!=(const FutureHandle& rhs) const { return !(*this == rhs); }
81 bool operator==(const FutureHandle& rhs) const {
82 // Only compare IDs, since the API is irrelevant (comparison will only occur
83 // within the context of a single API anyway).
84 return id() == rhs.id();
85 }
86
87 FutureHandleId id() const { return id_; }
88 detail::FutureApiInterface* api() const { return api_; }
89
90 // Detach from the FutureApi. This handle will no longer increment the
91 // Future's reference count. This is mainly used for testing, so that you can
92 // still look up the Future based on its handle's ID without affecting the
93 // reference count yourself.
94 void Detach();
95
96 // Called by CleanupNotifier when the API is being deleted. At this point we
97 // can ignore all of the reference counts since all Future data is about to be
98 // deleted anyway.
99 void Cleanup() { api_ = nullptr; }
100
101 private:
102 FutureHandleId id_;
103 detail::FutureApiInterface* api_;
105};
