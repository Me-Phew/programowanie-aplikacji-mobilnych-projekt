797 void set_string_value(char* value) {
798 size_t len = strlen(value);
799 if (len < kMaxSmallStringSize) {
800 Clear(static_cast<Type>(kInternalTypeSmallString));
801 strncpy(value_.small_string, value, len + 1);
802 } else {
803 set_mutable_string(std::string(value, len));
804 }
805 }

812 void set_string_value(const std::string& value) { set_mutable_string(value); }
813
823 void set_mutable_string(const std::string& value,
824 bool use_small_string = true) {
825 if (value.size() < kMaxSmallStringSize && use_small_string) {
826 Clear(static_cast<Type>(kInternalTypeSmallString));
827 strncpy(value_.small_string, value.data(), value.size() + 1);
828 } else {
829 Clear(kTypeMutableString);
830 *value_.mutable_string_value = value;
831 }
832 }

842 void set_mutable_blob(const void* src_data, size_t size_bytes) {
843 uint8_t* dest_data = new uint8_t[size_bytes]; // Will be deleted when
844 // `this` is deleted.

846 memcpy(dest_data, src_data, size_bytes);
847 }
848 Clear(kTypeMutableBlob);
849 set_blob_pointer(dest_data, size_bytes);
850 }
