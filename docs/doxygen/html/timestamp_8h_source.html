218template <typename Clock, typename Duration>
219std::chrono::time_point<Clock, Duration> Timestamp::ToTimePoint() const {
220 namespace chr = std::chrono;
221 using TimePoint = chr::time_point<Clock, Duration>;
222
223 // Saturate on overflow
224 const auto max_seconds = chr::duration_cast<chr::seconds>(Duration::max());
225 if (seconds_ > 0 && max_seconds.count() <= seconds_) {
226 return TimePoint{Duration::max()};
227 }
228 const auto min_seconds = chr::duration_cast<chr::seconds>(Duration::min());
229 if (seconds_ < 0 && min_seconds.count() >= seconds_) {
230 return TimePoint{Duration::min()};
231 }
232
233 const auto seconds = chr::duration_cast<Duration>(chr::seconds(seconds_));
234 const auto nanoseconds =
235 chr::duration_cast<Duration>(chr::nanoseconds(nanoseconds_));
236 return TimePoint{seconds + nanoseconds};
237}
