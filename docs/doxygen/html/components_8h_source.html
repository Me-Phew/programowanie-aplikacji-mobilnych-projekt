283 // For backwards compatibility with dynamic_link_domain, if

286 static const size_t kHttpsPrefixLength = sizeof(kHttpsPrefix) - 1;
287 if (strncmp(domain_uri_prefix, kHttpsPrefix, kHttpsPrefixLength) != 0) {
288 domain_uri_prefix_with_scheme =
289 std::string(kHttpsPrefix) + domain_uri_prefix;
290 domain_uri_prefix = domain_uri_prefix_with_scheme.c_str();
291 }
